#Without resolve_btfids the dkms module will not compile successfully:
#https://unix.stackexchange.com/questions/762985/compiling-external-kernel-module-fails-on-debian-bookworm-due-to-missing-resolve

Please reference the link and the instructions to pull the tool from the linux source and add to the system.

apt install linux-source-6.1
apt install locate
updatedb
locate  resolve_btfids
locate  linux-source-6.1.tar.xz

cd /usr/src/
tar xvf linux-source-6.1.tar.xz
find linux-source-6.1 | grep resolve_btfids
cd linux-source-6.1/tools/bpf/resolve_btfids
make
file resolve_btfids
ls /usr/src/linux-headers-`uname -r`/tools/bpf/
ls /usr/src/linux-headers-`uname -r`/tools/
mkdir -p /usr/src/linux-headers-`uname -r`/tools/bpf/resolve_btfids
ls /usr/src/linux-headers-`uname -r`/tools/
ls resolve_btfids
pwd
ln -s /usr/src/linux-source-6.1/tools/bpf/resolve_btfids/resolve_btfids /usr/src/linux-headers-`uname -r`/tools/bpf/resolve_btfids
